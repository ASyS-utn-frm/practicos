%------------------------------------
\subsection*{Pruebas de convergencia}

\textit{Prueba de la raz\'on}: Si para una serie dada $z_1+z_2+\cdots$ con $z_n \neq 0$ $(n=1,2,\cdots)$ se tiene que,
\begin{equation}\label{eq:pruebarazon}
\lim_{n \to \infty}\left|\dfrac{z_{n+1}}{z_n}\right|=L,
\end{equation}
entonces,
\begin{align*}
&\text{Si }L< 1 \text{, la serie converge absolutamente.}\\
&\text{Si }L> 1 \text{, la serie diverge.}\\
&\text{Si }L=1 \text{(o el limite no existe), la prueba fracasa y no puede concluirse nada..}\\
\end{align*}
la serie diverge. 

\textit{Prueba de la ra\'iz}: Si una serie $z_1+z_2+\cdots$ es tal que para toda $n$ mayor que alguna $N$,
\begin{equation}\label{eq:pruebaraiz}
\lim_{n \to \infty}\sqrt[n]{\left| z_n \right|} < 1,
\end{equation}
entonces, esta serie converge absolutamente. Si, en cambio, para una infinidad de $n$ se tiene que
\begin{equation}
\lim_{n \to \infty}\sqrt[n]{\left| z_n \right|} \geq 1,
\end{equation}
la serie diverge. 

%--------------------------------
\subsection*{Series de potencias}
Una {serie de potencias} de $z-z_0$ es una serie de la forma:
\begin{equation}\label{eq:seriepotencias}
\sum_{n=0}^{\infty} a_n (z-z_0)^n = a_0+a_1 (z-z_0) + a_2 (z-z_0)^2 + \cdots
\end{equation}
en donde $z$ es una variable, $a_0, a_1, \cdots$ son constantes, denominadas {coeficientes} de la serie, y $z_0$ es una 
constante, 
denominada {centro} de la serie. 

Toda serie de potencias que converge en su centro $z_0$, lo hace también para todo $z$ 
en un círculo de radio $R$ centrado en 
$z_0$. Esto es para todo $z$ que satisfaga $|z-z_0|<R$, donde $R$ puede ser hallado aplicando los limites dados por el 
teorema de la razón, Ec. \ref{eq:pruebarazon}, o de la raíz, Ec. \ref{eq:pruebaraiz}.

%\begin{equation}
%R=\frac{1}{L}.
%\end{equation}

Es importante destacar que toda función analítica puede ser representada mediante una serie de potencias como 
 determina el teorema de Taylor (ver abajo).
%----------------------------
\subsection*{Serie de Taylor}
Se denomina {serie de Taylor} de una función $f(z)$ analítica en un punto $z_0$ (centro de la serie) y su vecindad, 
a la serie de potencias dada por:
\begin{equation}\label{eq:serietaylor}
f(z) = \sum_{n=0}^\infty \dfrac{f^{(n)}(z_0)}{n!} (z-z_0)^n.
\end{equation}
El caso particular con centro $z_0=0$ se denomina {serie de Maclaurin} de $f(z)$.

%-------------------------------
\subsection*{Series generadoras}
Se utilizan para encontrar la representaci\'on en serie de otras funciones (notar que el centro de estas series 
es $z_0=0$).
\begin{align}
	\dfrac{1}{1-z} &= \sum_{n=0}^{\infty} z^n; 
	\qquad |z|<1													\label{eq:ht1} \\
	e^z &= \sum_{n=0}^{\infty} \dfrac{z^n}{n!}; 
	\qquad |z|<\infty												\label{eq:ht2} \\
	\cos z &= \sum_{n=0}^{\infty} (-1)^n\dfrac{z^{2n}}{(2n)!}; 
	\qquad |z|<\infty												\label{eq:ht3} \\
	\sin z &= \sum_{n=0}^{\infty} (-1)^n\dfrac{z^{2n+1}}{(2n+1)!}; 
	\qquad |z|<\infty												\label{eq:ht4}	
\end{align}


%----------------------------
\subsection*{Serie de Laurent}
La serie de Laurent generaliza a la serie de Taylor. Si se desea desarrollar una funci\'on $f(z)$ en potencias de 
$z-z_0$ cuando $f(z)$ es singular en $z_0$, no se puede usar la serie de Taylor. En su lugar se usa la llamada {serie 
de Laurent}, que consiste de potencias enteras positivas de $z-z_0$ tanto como de potencias enteras negativas de 
$z-z_0$.
Se denomina {serie de Laurent} de $f(z)$ \textit{con centro} $z_0$ a
\begin{equation}\label{eq:serielaurent}
f(z) = \sum_{n=-\infty}^\infty a_n(z-z_0)^n,~~~a_n=\frac{1}{2\pi i}\oint_{C}{\frac{f(z)}{(z-z_0)^{n+1}} dz},
\end{equation}
donde $C$ es una curva que contienen a $z_0$ (Prestar atenci\'on a que la sumatoria empieza en $-\infty$ en tanto que en la Ec. \ref{eq:seriepotencias} empieza en cero). Si $f(z)$ es no analítica en $z_0$, entonces al coeficiente $a_{-1}$ se le denomina Residuo de la serie de $f(z)$ con centro en $z_0$, o bien,
\begin{equation}\label{eq:residuo}
Res[f(z),z_0]=a_{-1}=\frac{1}{2\pi i}\oint_{C}{f(z)}dz.
\end{equation}

\textit{Nota importante:} No es com\'un utilizar la expresi\'on para $a_n$ dada en la Ec. \ref{eq:serielaurent} para 
encontrar un desarrollo en serie de Laurent. Lo que se hace es tratar de indentificar en la $f(z)$ la \textit{parte 
anal\'itica} de la funci\'on, desarrollar esa parte utilizando series generadoras o serie de Taylor correspondiente, y 
luego, mediante operaciones matem\'aticas, llegar a la expresi\'on en serie de Laurent. (Ver ejercicio de ejemplo para 
m\'as detalles)

\subsection*{Formulas para la obtención de residuos}
\textit{Dos f\'ormulas para obtener residuos de polos simples:}
La primer f\'ormula para obtener el residuo de una $f(z)$ con un polo simple en $z=z_0$ es,
\begin{equation}\label{eq:respolosimple1}
Res[f(z),z_0]=\lim_{z \to z_0} \left[(z-z_0) f(z)\right].
\end{equation}
Se podr\'a observar que la Ec. \ref{eq:respolosimple1} es la Ec. \ref{eq:respolosup} con $m=1$. La otra f\'ormula puede 
usarse cuando $f(z)$ es de la forma,
\begin{equation}
f(z)=\frac{p(z)}{q(z)},
\end{equation}
con $p(z)$ y $q(z)$ anal\'iticas, en donde se supone que $p(z_0)\neq 0$ y $q(z)$ tiene un cero simple en $z=z_0$. En 
este caso, la f\'ormula para obtener el residuo en un polo simple es entonces,
\begin{equation}\label{eq:respolosimple2}
Res[f(z),z_0]=\frac{p(z_0)}{q'(z_0)}.
\end{equation}

\textit{F\'ormula para el residuo de un polo de cualquier orden:} Si la funcion $f(z)$ tienen un polo de m-esimo orden 
en el punto $z=z_0$, entonces el residuo esta dado por,
\begin{equation}\label{eq:respolosup}
Res[f(z),z_0]=\frac{1}{(m-1)!}\lim_{z \to z_0} \left[\frac{d^{m-1}}{dz^{m-1}}\left[(z-z_0)^m f(z)\right]\right].
\end{equation}


%----------------------------
\subsection*{Teorema de los residuos}
Sea $f(z)$ una función analítica dentro y sobre una trayectoria cerrada $C$, excepto en un numero finito de puntos ${z_1,z_2,...,z_k}$ dentro de C, entonces
\begin{equation}\label{eq:resteorema}
\oint_{C}{f(z)}dz=2\pi i \sum_{j=1}^{k} Res[f(z),z_j].
\end{equation}

\subsection*{Resolución de integrales reales por el método de los residuos}
\textit{Integrales propias de funciones racionales de $\cos(\theta)$ y $\sin(\theta)$:}

Para resolver integrales propias de funciones racionales que involucren $\cos(\theta)$ y $\sen(\theta)$, primero 
transformamos 
la integral real a una compleja utilizando la definición de estas funciones (ver TP1) y luego realizamos el cambio de 
variables $z=e^{i\theta}$, notando que $dz/d\theta=iz$ y que, si $0<\theta<2\pi$, entonces la $z$ varia sobre la 
circunferencia $|z|<1$. Esto permite resolver la integral real mediante el teorema de los residuos. \\

\textit{Integrales impropias de funciones racionales:}

Se consideran las integrales impropias reales de la forma
\begin{equation}
\int_{-\infty}^{\infty}f(x)dx.
\end{equation}
Donde $f(x)$ no debe poseer polos sobre el eje $x$ y debe ser una función racional tal que, el grado del denominador 
sea 
dos o mas ordenes mayor que el grado del numerador.
B\'asicamente el m\'etodo consiste en reemplazar $x$ por $z$ para obtener una expresi\'on para $f(z)$ y resolver,
\begin{align}
\int_{-\infty}^{\infty}f(x)dx &= \oint_C f(z)dz, \\
&= 2\pi i\sum \text{Res} f(z),
\end{align}
en donde se suma sobre todos los residuos de $f(z)$ correspondientes a los polos de $f(z)$ en el \textit{semiplano 
superior}.
