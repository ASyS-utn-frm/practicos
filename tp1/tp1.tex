\subsection*{Representación analítica de un número complejo z}
Forma cartesiana (útil al trabajar con sumas algebraicas):
\begin{equation}
z = x+iy \to Re(z)=x=\frac{1}{2}(z+\bar z)~\wedge~Im(z)=y=\frac{1}{2i}(z-\bar z).
\label{eq:p1_ht1}
\end{equation}
Forma polar (útil la trabajar con productos y divisiones):
\begin{equation}
z = re^{i\theta} \to 
\mod(z)=|z|=r=\sqrt{x^2+y^2}~\wedge~\arg(z)=\theta=\arctan \Big(\frac{y}{x}\Big),
\label{eq:p1_ht2}
\end{equation}
\textbf{Nota:} Al calcular el $\arg{(z)}$, recordar que $\arctan$ es una función multivaluada (por ser la inversa de 
una función periódica). Por lo que debe interpretarse correctamente el resultado arrojado por la calculadora (solo es 
correcto en el primer y cuarto cuadrante). Frecuentemente resulta mas práctico interpretar el resultado de las 
funciones trigonométricas inversas si se usa la calculadora en modo Deg (grados sexagesimales).\\

Forma trigonométrica (útil la trabajar con sumas algebraicas). Utilizado la importante relación de Euler, podemos 
escribir:
\begin{equation}
z=r[\cos{(\theta)}+i\sin{(\theta)}] \to r\cos{(\theta)}=x~\wedge~r\sin{(\theta)}=y.
\label{eq:p1_ht3}
\end{equation}

\subsection*{Raíz enésima de un número complejo $z$}
\begin{equation}
w=\sqrt[n]{z} \to z=w^n \to r_ze^{i\theta_z}=\Big(r_we^{i\theta_w}\Big)^n=r_w^ne^{in\theta_w}.
\label{eq:p1_ht4}
\end{equation}
Igualando complejos se obtiene la expresión para calcular la raíz ($w$) en forma polar,
\begin{equation}
r_w=\sqrt[n]{r_z}~\wedge~\theta_w=\frac{\theta_z+2k\pi}{n},~k=0 ,1,2,...,n-1
\label{eq:p1_ht5}
\end{equation}
\textbf{Nota:} La raíz enésima es una función multivaluada con infinitas soluciones pero con solo $n$ soluciones fundamentales distintas. Estas están distribuidas en el 
plano complejo de manera equidistante 
sobre la circunferencia de radio $r_w$.

\subsection*{Potencia real de un número complejo $z$}
\begin{equation}
w=z^n=\Big(r_ze^{i\theta_z}\Big)^n=r_z^ne^{in\theta_z}.
\label{eq:p1_ht41}
\end{equation}
Igualando complejos se obtiene la expresión para calcular la potencia ($w$) en forma polar,
\begin{equation} 
r_w=r_z^n~\wedge~\theta_w=n\theta_z+2k\pi,~k=0,1,2,...
\label{eq:p1_ht51}
\end{equation}

\subsection*{Funciones trigonométricas e hiperbólicas}

Se presenta una serie de fórmulas de utilidad que serán usadas en posteriores desarrollos.

\begin{equation}\label{trigo_cos_e}
\cos z = \frac{1}{2}(e^{iz}+e^{-iz}).
\end{equation}

\begin{equation}\label{trigo_sin_e}
\sin z = \frac{1}{2i}(e^{iz}-e^{-iz}).
\end{equation}

\begin{equation}\label{trigo_cos_z}
\cos z = \cos x \cosh y - i \sin x \sinh y.
\end{equation}

\begin{equation}\label{trigo_sin_z}
\sin z = \sin x \cosh y + i \cos x \sinh y.
\end{equation}

\begin{equation}\label{trigo_cos_sin}
\cos ^2 z + \sin ^2 z = 1.
\end{equation}

\begin{equation}\label{hiper_cos_e}
\cosh z = \frac{1}{2}(e^{z}+e^{-z}).
\end{equation}

\begin{equation}\label{hiper_sin_e}
\sinh z = \frac{1}{2i}(e^{z}-e^{-z}).
\end{equation}

\begin{equation}\label{hiper_cos_z}
\cosh z = \cosh x \cos y + i \sinh x \sin y.
\end{equation}

\begin{equation}\label{hiper_sin_z}
\sinh z = \sinh x \cos y + i \cosh x \sin y.
\end{equation}

\begin{equation}\label{hiper_cos_sin}
\cosh ^2 z - \sinh ^2 z = 1.
\end{equation}

\subsection*{Logaritmo de un número complejo $z$}
\begin{equation}
w=\ln{(z)}=\ln{(r_ze^{i\theta_z})}=\ln{(r_z)}+i(\theta_z+2k\pi),~k=0,1,2,...
\label{eq:p1_ht6}
\end{equation}
Igualando complejos se obtiene la expresión para calcular el logaritmo ($w$) en forma cartesiana,
\begin{equation}
\operatorname{Re}(w)=ln{(r_z)}~\wedge~\operatorname{Im}{(w)}=\theta_z+2k\pi,~k=0,1,2,...
\label{eq:p1_ht7}
\end{equation}

\subsection*{Potencia compleja de un numero real}
\begin{equation}
w=A^z\to w=e^{\ln{(A)}z}=e^{\ln{(A)}x} e^{i\ln{(A)}y}
\label{eq:p1_ht8}
\end{equation}
Igualando complejos se obtiene la expresión para calcular la potencia ($w$) en forma polar,
\begin{equation}
r_w=e^{\ln{(A)}x}~\wedge~\theta_w=\ln{(A)y}+2k\pi,~k=0,1,2,...
\label{eq:p1_ht9}
\end{equation}
Notar que si la base es $A=e$ la resolución es directa:
\begin{equation}
w=e^{z}=e^{x+iy}=e^xe^{iy} \to r_w=e^x \wedge \theta_w=y+2k\pi,~k=0,1,2,...
\label{eq:p1_ht10}
\end{equation}
