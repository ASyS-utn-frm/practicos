%% Derivada
\subsection*{Derivada}
La derivada de una función compleja $f$ en un punto $z_0$ se define como
\begin{equation}
f'(z_0)=\lim_{{\Delta z}\to{0}}\frac{f(z)-f(z_0)}{z-z_0}.
\end{equation}
Esta definición es como la definición de derivada en cálculo real. Por lo que las reglas de diferenciación son las 
mismas, y pueden usarse acá para encontrar la derivada de una función compleja.

A la función compleja $f(z)$ también la podemos escribir
\begin{equation}
f(z)=u(x,y)+iv(x,y),
\end{equation}
donde $u(x,y)$ y $v(x,y)$ son funciones reales que dependen de las variables reales $x$ e $y$, siendo $z=x+iy$. 

Las derivadas parciales de una función compleja $f(z)=f(x,y)$ pueden escribirse como:
\begin{equation}\label{eq:deri_parcial}
\frac{\partial f(x,y)}{\partial x}=\frac{\partial u}{\partial x}+i\frac{\partial v}{\partial x}, \qquad
\frac{\partial f(x,y)}{\partial y}=\frac{\partial v}{\partial y}-i\frac{\partial u}{\partial y}.
\end{equation}

%% Ecuaciones de Cauchy-Riemann
\subsection*{Ecuaciones de Cauchy-Riemann}
Las ecuaciones de Cauchy-Riemann permiten verificar la analiticidad de una función compleja. En términos generales, $f$ 
es analí­tica en un dominio $D$ si y sólo si las primeras derivadas parciales de $u$ y $v$ satisfacen las dos ecuaciones
\begin{equation}
\label{eq:cauchy-riemann}
\frac{\partial u}{\partial x} = \frac{\partial v}{\partial y}, \qquad \qquad
\frac{\partial u}{\partial y} = -\frac{\partial v}{\partial x}.
\end{equation}
Las Ec. \ref{eq:cauchy-riemann} son las ecuaciones de Cauchy-Riemann.

Si se usa la forma polar $z=r(\cos\theta+i\sin\theta)$ y se hace $f(z)=u(r,\theta)+iv(r,\theta)$, entonces las 
ecuaciones de Cauchy-Riemann en forma polar son
\begin{equation}
\frac{\partial u}{\partial r} = \frac{1}{r}\frac{\partial v}{\partial \theta}, \qquad \qquad
\frac{\partial v}{\partial r} = -\frac{1}{r}\frac{\partial u}{\partial \theta}.
\end{equation}

%% Ecuación de Laplace
\subsection*{Ecuación de Laplace. Funciones armónicas.}
Una funci\'on que satisface la ecuaci\'on de Laplace $\nabla^2 u = 0$ en cierta regi\'on, se dice que es 
\textit{arm\'onica} en esa regi\'on.

Si $f(z)=u(x,y)+iv(x,y)$ es anal\'itica en un dominio $D$, entonces $u$ y $v$ satisfacen la ecuaci\'on de Laplace.
	\begin{align}
	\label{eq:ht_laplace}
		\nabla^2 u &= \frac{\partial^2u}{\partial x^2} + \frac{\partial^2u}{\partial y^2} =0,&
		\nabla^2 v &= \frac{\partial^2v}{\partial x^2} + \frac{\partial^2v}{\partial y^2} =0.
	\end{align}

Si dos funciones arm\'onicas $u$ y $v$ satisfacen las ecuaciones de Cauchy-Riemann en un dominio $D$, entonces son las 
partes real e imaginaria de una funci\'on anal\'itica $f$ en $D$. Entonces se dice que $v$ es la \textit{funci\'on 
arm\'onica conjugada} de $u$ en $D$.

\textit{Forma cartesiana}

Primero se reemplaza $z=x+iy$, y se opera para separar en parte real e imaginaria
\begin{align}
f(x+iy)	&= \frac{1}{x+iy}, \\
		&= \frac{1}{x+iy}\cdot\frac{x-iy}{x-iy}, \\
		&= \frac{x}{x^2+y^2}-i\frac{y}{x^2+y^2}.
\end{align}
De esto se obtiene
\begin{equation}
u(x,y)=\frac{x}{x^2+y^2}, \qquad\qquad v(x,y)=-\frac{y}{x^2+y^2}.
\end{equation}
Se realizan las derivadas parciales
\begin{equation}
\frac{\partial u}{\partial x} = \frac{x^2+y^2-x2x}{(x^2+y^2)^2} = \frac{y^2-x^2}{(x^2+y^2)^2},
\end{equation}
\begin{equation}
\frac{\partial v}{\partial y} = \frac{(-1)(x^2+y^2)-(-y)2y}{(x^2+y^2)^2} = \frac{-x^2-y^2+2y^2}{(x^2+y^2)^2} = 
\frac{y^2-x^2}{(x^2+y^2)^2}.
\end{equation}
Se cumple la primer ecuación $\frac{\partial u}{\partial x} = \frac{\partial v}{\partial y}$. Se sigue el desarrollo,
\begin{equation}
\frac{\partial v}{\partial x} = \frac{-(-y)2x}{(x^2+y^2)^2} = \frac{2xy}{(x^2+y^2)^2},
\end{equation}
\begin{equation}
\frac{\partial u}{\partial y} = \frac{-x2y}{(x^2+y^2)^2} = \frac{-2xy}{(x^2+y^2)^2}.
\end{equation}
Se cumple la segunda ecuación $\frac{\partial v}{\partial x} = -\frac{\partial u}{\partial y}$, por lo que se concluye 
que $f(z)$ es analí­tica, salvo en $z=0$.

\textit{Forma polar}

Se reemplaza $z=re^{i\theta}$,
\begin{equation}
f(r,\theta)=\frac{1}{re^{i\theta}}=\frac{1}{r}e^{-i\theta}=\frac{1}{r}\cos \theta-i\frac{1}{r}\sin \theta.
\end{equation}
Se realizan las derivadas parciales,
\begin{equation}
\frac{\partial u}{\partial r} = \frac{-\cos \theta}{r^2},
\end{equation}
\begin{equation}
\frac{\partial v}{\partial \theta} = \frac{-\cos \theta \cdot r}{r^2} = \frac{-\cos \theta}{r},
\end{equation}
lo que verifica la ecuación $\frac{\partial u}{\partial r} = \frac{1}{r}\frac{\partial v}{\partial \theta}$. Seguimos,
\begin{equation}
\frac{\partial v}{\partial r} = \frac{-(-\sin \theta)}{r^2}=\frac{\sin \theta}{r^2},
\end{equation}
\begin{equation}
\frac{\partial u}{\partial \theta} = \frac{-\sin \theta \cdot r}{r^2} = \frac{-\sin \theta}{r},
\end{equation}
Que verifica la ecuación $\frac{\partial v}{\partial r} = -\frac{1}{r}\frac{\partial u}{\partial \theta}$. Verificando 
nuevamente que $f(z)$ es analí­tica.

	\begin{equation}

	\end{equation}

	\begin{equation}
		\int 1 dy = y + h(x) = v(x,y).
	\end{equation}

	\begin{equation}
		\dfrac{\partial v}{\partial x} = \dfrac{\partial h(x)}{\partial x}.
	\end{equation}

	\begin{equation}
		\dfrac{\partial h(x)}{\partial x} = - \dfrac{\partial u}{\partial y} = 0

	\begin{equation}
		\dfrac{\partial h(x)}{\partial x} = 0 ~~~ \Rightarrow ~~~ h(x) = C.
	\end{equation}

	\begin{equation}
		v(x,y) = y + C.
	\end{equation}
	
	Finalmente,
	\begin{equation}
		\boxed{f(x,y) = u(x,y) + i v(x,y) = x + i (y+C) = x + iy + iC = z + iC}.
	\end{equation}

	\item $ u(x,y) =e^{2x}\cos{(ay)}$
	
	Para que $u(x,y)$ sea arm\'onica, debe satisfacer la ecuaci\'on de Laplace:

	\begin{equation}\label{parcialux5.a}
	\dfrac{\partial u}{\partial x} =2e^{2x}\cos{(ay)} ; ~~~~ \dfrac{\partial^2 u}{\partial x^2} = 4e^{2x}\cos{(ay)}.
	\end{equation}
	\begin{equation}\label{parcialuy5.a}
	\dfrac{\partial u}{\partial y} = -e^{2x}\sin{(ay)a} ; ~~~~ \dfrac{\partial^2 u}{\partial y^2} = 
	-a^2e^{2x}\cos{(ay)}.
	\end{equation}
	
	La funci\'on es arm\'onica si se anula su Laplaciano (Ec.\ref{eq:ht_laplace}), es decir si la suma de las derivadas 
	segundas en Ec. 
	\ref{parcialux5.a} y \ref{parcialuy5.a} es nula :
	
	\begin{align}
	4e^{2x}\cos{(ay)}-a^2e^{2x}\cos{(ay)}=0 \\
	(4-a^2)e^{2x}\cos{(ay)}=0\\
	(4-a^2)=0\\
	a=\pm2
	\end{align}	

	Para encontrar la funci\'on anal\'itica correspondiente, hacemos uso de las ecuaciones de Cauchy-Riemann, ver Ec. 
	\ref{eq:cauchy-riemann}:
	
	\begin{equation}
	\dfrac{\partial u}{\partial x} = \dfrac{\partial v}{\partial y} = 2e^{2x}\cos{(ay)}.
	\end{equation}
	
	Con el objetivo de encontrar $v(x,y)$, integramos la expresión anterior con respecto a $y$,
	\begin{equation}
	v(x,y)=\int 2e^{2x}\cos{(ay)}~dy =2e^{2x} \int \cos{(ay)}~dy=\frac{2e^{2x}}{a}\sin{(ay)} + h(x).
	\end{equation}
	
	Ahora derivamos $v(x,y)$ con respecto a $x$,
	\begin{equation}
	\dfrac{\partial v}{\partial x} = \frac{4e^{2x}}{a}\sin{(ay)}+\dfrac{\partial h(x)}{\partial x}.
	\end{equation}
	
	Por la condici\'on de Cauchy-Riemann, Ec. \ref{eq:cauchy-riemann}, debe ser,
	\begin{equation}
	 \frac{4e^{2x}}{a}\sin{(ay)}+\dfrac{\partial h(x)}{\partial x} = - \dfrac{\partial u}{\partial y} = 
	 e^{2x}\sin{(ay)a}
	\end{equation}
	
	Comparando primer y tercer miembro, y reconrdando que $a=\pm2$, obtenemos,
	\begin{equation}
	\dfrac{\partial h(x)}{\partial x} = 0 ~~~ \Rightarrow ~~~ h(x) = \int 0~dx=C.
	\end{equation}
	
	Escribimos entonces la función armónica conjugada,
	\begin{equation}
	\boxed{v(x,y) = e^{2x}\sin{(2y)} +C.}
	\end{equation}
	
	
\end{enumerate}
