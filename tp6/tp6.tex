\section*{Herramientas Te\'oricas:}
\label{sec:herr}

\subsection*{Linealidad}
Un sistema es \textit{lineal} si posee la propiedad de \textit{superposici\'on}, que deriva de las siguientes dos propiedades. \\
La propiedad de \textbf{aditividad}, que se enuncia de la siguiente manera,
\begin{align}
x_1(t) &\to y_1(t), \\
x_2(t) &\to y_2(t), \\
x_1(t)+x_2(t) &\to y_1(t)+y_2(t).
\end{align}
Y la propiedad de \textbf{homogeneidad}, o escalamiento, que se enuncia de la siguiente manera,
\begin{align}
x_1(t) &\to y_1(t), \\
ax_1(t) &\to ay_1(t).
\end{align}
Las dos propiedades combinadas resultan en,
\begin{equation}
a_1x_1(t)+a_2x_2(t) \to a_1y_1(t)+a_2y_2(t).
\end{equation}
De forma general, la propiedad de superposici\'on es,
\begin{equation}
\sum_k a_kx_k(t) \to \sum_k a_ky_k(t).
\end{equation}
Notar que un sistema es lineal siempre que las operaciones matemáticas que lo describen son lineales.
\subsection*{Causalidad}
Un sistema es causal si su salida depende unicamente de valores presentes y pasados de la entrada. Usualmente se
demuestra la no causalidad del sistema mediante un ejemplo. 

\subsection*{Invariancia en el tiempo}
Un sistema es invariante en el tiempo si para una entrada desplazada en el tiempo, le corresponde la salida 
igualmente desplazada, es decir:
\begin{align}
\label{eq:ti}
x(t) &\to y(t), \\
x(t-t_0) &\to y(t-t_0).
\label{eq:ti2}
\end{align}
La invariancia en el tiempo esta relacionada a que la función que define la acción del sistema no incluye
escalado temporal y presenta términos multiplicativos o aditivos constantes.

\subsection*{Sistema LTI}
Un sistema es LTI (del Inglés, Linear Time Invariant) si es lineal e invariante en el tiempo. Esta clase de sistemas 
es de fundamental importancia en ingeniería debido a que permiten modelar sistemas físicos prácticos de diversa 
naturaleza (sistemas mecánicos, circuitos eléctricos, etc.). La salida de un sistema LTI, $y(t)$, para una entrada 
genérica, $x(t)$, puede calcularse conociendo únicamente su respuesta impulsiva, $h(t)$, mediante $y(t)=x(t)*h(t)$.

\subsection*{Integral de convolución}
La integral de convolución se representa y define como
\begin{equation}\label{eq:intconv}
y(t)=x(t)*h(t)=\int_{-\infty}^{\infty}x(\tau)h(t-\tau)d\tau.
\end{equation} 

\subsection*{Suma de convolución}
La suma de convolución se representa y define como
\begin{equation}
y[n]=x[n]*h[n]=\sum_{k=-\infty}^\infty x[k]h[n-k].
\end{equation}

\subsection*{Estabilidad}
La estabilidad de un sistema puede determinarse según la respuesta al impulso de dicho sistema.

Para tiempo discreto, si la respuesta al impulso unitario es absolutamente sumable, entonces $y[n]$ está limitada en magnitud y por consiguiente el sistema es estable. Esto se escribe,
\begin{equation}
\sum_{k=-\infty}^{+\infty} |h[k]|<\infty.
\end{equation}
Esta es condición suficiente para garantizar la estabilidad de un sistema LIT de tiempo discreto.

De la misma manera, para sistemas de tiempo continuo, un sistema es estable si la respuesta al impulso es absolutamente integrable, esto es,
\begin{equation}
\int_{-\infty}^{+\infty}|h(\tau)|d\tau <\infty
\end{equation}

\item[\textit{a})] 
	Para una entrada $x_1(t)$ en el sistema, se obtiene,
	\begin{equation}
	x_1(t) \to y_1(t)=\cos [x_1(t)],
	\end{equation}
	y de la misma manera para una $x_2(t)$,
	\begin{equation}
	x_2(t) \to y_2(t)=\cos [x_2(t)].
	\end{equation}
	Para probar si se cumple la propiedad de aditividad, consideramos una se\~nal $x(t)=x_1(t)+x_2(t)$, con lo que se 
	tiene,
	\begin{equation}
	x(t) \to y(t)=\cos [x(t)]=\cos [x_1(t)+x_2(t)].
	\end{equation}
	Comparando, se observa que,
	\begin{equation}
	y(t)=\cos [x_1(t)+x_2(t)] \neq y_1(t) + y_2(t).
	\end{equation}
	Se concluye entonces que el sistema \boxed{\text{no es lineal}}.
\end{profesor}
% --- (b) ---
\item[\textit{b})] 
Para una entrada $x_1(t)$ en el sistema, se obtiene,
\begin{equation}
x_1(t) \to y_1(t)=t^2 x_1(t-1),
\end{equation}
y de la misma manera para una $x_2(t)$,
\begin{equation}
x_2(t) \to y_2(t)=t^2 x_2(t-1).
\end{equation}
Para probar si se cumple la propiedad de aditividad, consideramos una se\~nal $x(t)=x_1(t)+x_2(t)$, con lo que se tiene,
\begin{align}
x(t) \to y(t) 	&= t^2 x(t-1), \\
				&= t^2 [x_1(t-1)+x_2(t-1)], \\
				&= t^2 x_1(t-1) + t^2 x_2(t-1), \\
				&= y_1(t) + y_2(t).
\end{align}
De lo anterior se observa que se cumple la propiedad de aditividad. Es necesario verificar ahora si se cumple la propiedad de homogeneidad.
Se toma una entrada $x(t)=kx_1(t)$, con la que se obtiene,
\begin{align}
x(t) \to y(t) 	&= t^2 x(t-1), \\
				&= t^2 [kx_1(t-1)], \\
				&= k t^2 x_1(t-1), \\
				&= ky_1(t).
\end{align}
Se ve finalmente que el sistema cumple con la propiedad de homogeneidad, por lo que se concluye que el sistema 
\boxed{\text{sí es lineal}}.
\end{enumerate}

\item Reemplazando las expresiones dadas para las señales $h(t)=e^{-t}u(t)u(5-t)$ y $x(t)=tu(t)u(3-t)$ en la Ec. 
\ref{eq:intconv} se obtiene,
\begin{equation}
y(t)=\int_{-\infty}^{\infty} \tau u(\tau)u(3-\tau) e^{-t+\tau}u(t-\tau)u(5-t+\tau) d\tau
\end{equation}
Notar que en estos tramos el producto de los escalones en el integrando valdrá 0 o 1. La resolución de la integral de 
convolución debe hacerse para los tramos en donde el integrando no vale 0. Por lo que primeramente resolvemos la 
integral para dos limites genéricos $a$ y $b$ que dependerán del tramo considerado:

\begin{align}
y(t)&=\int_{a}^{b} \tau e^{-t+\tau} d\tau \\
&=e^{-t}\int_{a}^{b} \tau e^{\tau} d\tau
\end{align}
Resolvemos por partes con $u=\tau,~dv=e^{\tau}d\tau,~du=d\tau,~v=e^{\tau}$:
\begin{align}
y(t)&=e^{-t}\Big[ \tau e^{\tau}|_a^b-\int_{a}^{b} e^{\tau}  d\tau\Big] \\
&=e^{-t}\Big[ \tau e^{\tau}|_a^b-e^{\tau}|_a^b\Big]\\
&=e^{-t}\Big[e^{\tau}(\tau-1)\Big]_a^b
\label{eq:ej7}
\end{align}

En la Fig. \ref{fig:ej7a}) se ve que ambas señales son de duración finita. Para derivar los intervalos de integración 
de los diferentes tramos en los que el integrando es no nulo, lo mas conveniente es realizar un gráfico:

\begin{itemize}
	\item Para $t<0$: Como puede verse en la Fig. \ref{fig:ej7b} no hay solapamiento, por lo que el producto de las 
	señales en el integrando es nulo: 
	\begin{equation}
	y_1(t)=0~~\forall~~t<0
	\end{equation}

	\item Para $0\leq t \leq 3$: Como puede verse en la Fig. \ref{fig:ej7c} hay solapamiento. Este tramo se extiende 
	desde que la $h(t)$ rebatida y desplazada toca a $x(t)$ en $t=0$ hasta que hay un solapamiento completo en $t=3$. 
	Para cada $t$ en este intervalo los limites de integración son de $0$ a $t$ por lo que la Ec. \ref{eq:ej7} queda:
	\begin{align}
	y_2(t)&=e^{-t}\Big[e^{\tau}(\tau-1)\Big]_0^t\\
	&=e^{-t}\Big[e^{t}(t-1)-e^{0}(0-1)\Big]\\
	&=t-1+e^{-t}~~\forall~~0\leq t \leq 3 \\
	&=(t-1+e^{-t})u(t)u(-t+3)
	\end{align}

	\item Para $3< t \leq 5$: Como puede verse en la Fig. \ref{fig:ej7d} hay solapamiento. Este tramo se extiende 
	desde que la $h(t)$ rebatida y desplazada cubre completamente a $x(t)$ en $t=3$ hasta que deja de haber un 
	solapamiento completo en $t=5$. 
	Para cada $t$ en este intervalo los limites de integración son de $0$ a $3$ por lo que la Ec. \ref{eq:ej7} queda:
	\begin{align}
	y_3(t)&=e^{-t}\Big[e^{\tau}(\tau-1)\Big]_0^3\\
	&=e^{-t}\Big[e^{3}(3-1)-e^{0}(0-1)\Big]\\
	&=e^{-t}(2e^3+1)~~\forall~~3< t \leq 5 \\
	&=e^{-t}(2e^3+1)u(t-3)u(-t+5)
	\end{align}

	\item Para $5< t \leq 8$: Como puede verse en la Fig. \ref{fig:ej7e} hay solapamiento. Este tramo se extiende 
	desde que la $h(t)$ rebatida y desplazada deja de solaparse completamente con $x(t)$ en $t=5$ hasta que deja de 
	haber solapamiento  en $t=8$. 
	Para cada $t$ en este intervalo los limites de integración son de $t-5$ a $0$ por lo que la Ec. \ref{eq:ej7} queda:
	\begin{align}
	y_4(t)&=e^{-t}\Big[e^{\tau}(\tau-1)\Big]_{t-5}^3\\
	&=e^{-t}\Big[e^{3}(3-1)-e^{t-5}(t-5-1)\Big]\\
	&=2e^{3-t}-te^{-5}+6e^-5~~\forall~~5< t \leq 8\\
	&=\Big[2e^{3-t}-te^{-5}+6e^-5\Big]u(t-5)u(-t+8)
	\end{align}

	\item Para $t>8$: Como puede verse en la Fig. \ref{fig:ej7f} no hay solapamiento, por lo que el producto de las 
	señales en el integrando es nulo: 
	\begin{equation}
	y_5(t)=0~~\forall~~t>8
	\end{equation}

Finalmente, la convolución buscada se muestra en la  Fig. \ref{fig:ej7g} y puede escribirse como:
\begin{equation}
y(t)=\boxed{y_2(t)+y_3(t)+y_4(t)}
\end{equation}

\end{itemize}

\item 
\begin{enumerate}
\item
Se observa qué sucede cuando se conectan dos sistemas en cascada. 

Se toma en primer lugar una señal de salida $y_1[n]$ para el sistema cuya respuesta al impulso es $h_1[n]$. Esta salida es,
\begin{equation}
y_1[n]=x[n]*h_1[n],
\end{equation}
y tomando en cuenta esto, la salida del sistema cuya respuesta al impulso es $h_2[n]$ es,
\begin{equation}
y[n]=y_1[n]*h_2[n]=\boxed{x[n]*h_1[n]*h_2[n]}.
\end{equation}
Lo anterior está relacionado con la propiedad asociativa de sistemas, que nos dice básicamente que dos sistemas conectados en cascada, con respuesta impulsiva $h_1[n]$ y $h_2[n]$, equivalen a un único sistema con respuesta impulsa dada por $h_1[n]*h_2[n]$.

\begin{profesor}
\item
Se observa qué sucede cuando se conectan dos sistemas en paralelo.

Llegan al punto de suma las señales de salida de los sistemas con respuesta al impulso $h_1[n]$ y $h_2[n]$, lo que permite escribir a la señal de salida como,
\begin{equation}
y[n]=x[n]*h_1[n]+x[n]*h_2[n]=x[n]*\lbrace h_1[n]+h_2[n]\rbrace,
\end{equation}
lo que indica que la combinación en paralelo de dos sistemas está asociado con la propiedad distributiva, resultando los dos sistemas en paralelo en un único sistema con respuesta al impulso dada por $h_1[n]+h_2[n]$.
\end{profesor}
\end{enumerate}
