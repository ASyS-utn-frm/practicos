\end{enumerate}

\tabto{9cm} Rta.: $y(t)=2e^{-2t}[\cos t + 8\sin t]u(t)$
\end{enumerate}

\subsection*{Transformada de Laplace (TL)}
La transformada bilateral (notar los limites de integración) de Laplace se define como: 
\begin{equation}
	X(s)=\mathcal{L}\{x(t)\}=\int_{-\infty}^{\infty}x(t)e^{-st} dt.
	\label{ec:tl}
\end{equation}
Notar que la Ec. \ref{ec:tl} es una transformada integral, que generaliza a la transformada de Fourier pues  
$s=\sigma + j\omega$. Dada esta definición de las variables transformadas, puede verse que $X(s)=X(j\omega)$ para 
$\sigma=0$. Para que la integral impropia de la Ec. \ref{ec:tl} converja absolutamente, el modulo del integrando 
$|x(t)e^{-st}|=|x(t)e^{-\sigma t}e^{-j\omega t}|=|x(t)|e^{-\sigma t}$, debe ser absolutamente integrable. Por lo que en 
general, $X(s)$ existirá únicamente para algunos valores o intervalos de $\sigma$ (bandas verticales o semiplanos en el 
plano $s$). Frecuentemente $X(s)$ será una función racional de polinomios en $s$, con sus polos determinando los 
limites de la región de convergencia.

Obviamos la definición de la TL inversa, pues para realizar la operación de anti-transformación, suelen 
utilizarse las tablas de pares de TL. 

\subsection*{Resolución de ecuaciones diferenciales mediante la transformada de Laplace}
La propiedad de derivación de la TL establece que si $X(s)=\mathcal{L}\{x(t)\}$, entonces:
\begin{equation}
	\mathcal{L}\Big\{\frac{dx(t)}{dt}\Big\}=sX(s)-x(0).
\end{equation}
Notar que esta propiedad simplifica considerablemente la resolución de ecuaciones diferenciales lineales a coeficientes 
constantes (las cuales son el modelo matemático de los sistemas LTI), pues la reduce a la resolución de una ecuación 
algebraica en el plano $s$.

De forma general, la fórmula de la transformada de Laplace de la derivada de cualquier orden $n$ es,
\begin{equation}
	\mathcal{L}\left\{\frac{d^{(n)}x(t)}{dt^{(n)}}\right\}=
	s^nX(s)-s^{n-1}x(0)-s^{n-2}x'(0)-\cdots-x^{(n-1)}(0).
\end{equation}

\subsection*{Función de transferencia de sistemas LTI}
La propiedad de convolución de la TL  establece que si $X(s)=\mathcal{L}\{x(t)\}$, $H(s)=\mathcal{L}\{h(t)\}$ y 
$Y(s)=\mathcal{L}\{y(t)\}$, entonces:
\begin{equation}
\mathcal{L}\Big\{x(t)*h(t)\Big\}=X(s)H(s).
\end{equation}
Notar que esta propiedad reduce el cálculo de la operación de convolución a una simple multiplicación en el plano $s$, 
y 
permite el cálculo de la operación inversa (deconvolución). 
Dado que la salida $y(t)$ de un sistema LTI con respuesta impulsiva $h(t)$ a una entrada $x(t)$ es: $y(t)=x(t)*h(t)$, 
la TL simplifica la obtención de $y(t)$. Además, la $H(s)$ se denomina \textit{funcion de transferencia del sistema}, 
y describe completamente el funcionamiento de todo sistema LTI ya que es una generalización de la respuesta en 
frecuencia del mismo ($H(j\omega)=H(s)$ para $\sigma=0$).

\subsection*{Desarrollo de funciones racionales en fracciones simples}
La TL inversa se realiza en la práctica mediante tablas de pares transformados. Para esto, generalmente 
se requiere operar sobre la expresión que se desea anti-transformar, a fin de expresarla en términos de funciones 
presentes en la tabla. Debido a que muchas de las funciones transformadas que surgen al tratar con sistemas LTI son 
cocientes de 
polinomios en $s$, una herramienta muy útil es la descomposición en fracciones simples.

Dada la expresión racional,
\begin{equation}
H(s)=\frac{Y(s)}{X(s)},
\end{equation}
donde $Y(s)$ y $X(s)$ son polinomios en $s$ sin factores comunes entre sí, y con el grado de  $Y(s)$ menor que el de  
$X(s)$. Podemos expresarla como una suma de fracciones simples siguiendo las reglas que se detallan a continuación. 

Según el tipo de factor presente en $X(s)$ tenemos:
\begin{itemize}
	\item[Caso 1:] Al factor $(s-a)$ le corresponde la fracción:
	\begin{equation}
		\frac{A}{s-a}.
	\end{equation} 
	Donde el coeficiente se calcula según:
	\begin{equation}
		A=\lim_{s\to a}\Big[(s-a)\frac{Y(s)}{X(s)}\Big].
	\end{equation}
			
	\item[Caso 2:] Al factor $(s-a)^m$  le corresponde la suma de fracciones:
	\begin{equation}
	\frac{A_m}{(s-a)^m}+\frac{A_{m-1}}{(s-a)^{m-1}}+...+\frac{A_1}{(s-a)}.
	\end{equation} 
	Donde el primer coeficiente $A_m$ se calcula según:
	\begin{equation}
	A_m=\lim_{s\to a}\Big[(s-a)^m\frac{Y(s)}{X(s)}\Big],
	\end{equation}
	y el resto de los coeficientes $A_k$ según:
	\begin{equation}
	A_k=\frac{1}{(m-k)!}\lim_{s\to a}\frac{d^{m-k}}{ds^{m-k}}\Big[(s-a)^m\frac{Y(s)}{X(s)}\Big]~\text{para}~k=1,...m-1.
	\end{equation}	
	\item[Caso 3:] Si hallamos el factor $(s-a)(s-\bar{a})$ en el denominador, no aplicamos el método de fracciones 
	simples pues es mas sencillo reemplazar:
	\begin{equation}
	\frac{Y(s)}{(s-a)(s-\bar{a})}=\frac{Y(s)}{(s-\alpha)^2+\beta^2},
	\end{equation}	
	donde $a=\alpha+j\beta$. A esta expresión puede llegarse también completando cuadrados en el denominador. Luego 
	notamos que el denominador en el miembro derecho puede ser antitransformado utilizando la tabla, como 
	$e^{-\alpha t}cos(\omega t)u(t)$ o $e^{-\alpha t}sin(\omega t)u(t)$, dependiendo de la forma de $Y(s)$.
	\end{itemize}

\subsection*{Estabilidad de sistemas LTI}
Si un sistema LTI es estable, entonces existe la TF de su respuesta impulsiva $\mathcal{F}\{h(t)\}=H(j\omega)$ 
(respuesta en frecuencia). Por lo que la región de convergencia de su función de transferencia, 
$\mathcal{L}\{h(t)\}=H(s)$, debe contener 
al eje imaginario ($\sigma=0$).   

\subsection*{Causalidad de sistemas LTI}
Si un sistema LTI es causal, entonces la región de convergencia de su función de transferencia,  
$\mathcal{L}\{h(t)\}=H(s)$, es un semiplano 
derecho. Si el sistema es no causal, entonces la región es un semiplano izquierdo.

	\item Transformado ambos miembros obtenemos la función de transferencia,
	\begin{align}
		s^2Y(s)+3sY(s)+Y(s)&=-\frac{1}{2}X(s),\\
		H(s)&=\frac{Y(s)}{X(s)}=\boxed{\frac{-1/2}{s^2+3s+1}}
	\end{align}
	Factorizando el polinomio en el denominador, $s^2+3s+1=(s+2.62)(s+0.38)$, obtenemos dos polos simples. Uno en -2.62 
	y el otro en -0.38.
	\item Para hallar $h(t)$, desarrollamos $H(s)$ usando fracciones simples,
	\begin{align}
		H(s)&=\frac{-1/2}{(s+2.62)(s+0.38)},\\
		&=\frac{A}{s+2.62}+\frac{B}{s+0.38},\\
		A &=\frac{-1/2}{s+0.38}\Bigg |_{s=-2.62}=0.22,\\	
		B &=\frac{-1/2}{s+2.62}\Bigg |_{s=-0.38}=-0.22,\\		
		H(s)&=\frac{0.22}{s+2.62}-\frac{0.22}{s+0.38}.	
	\end{align}
    Luego obtenemos $H(j\omega)$, al hacer $s=j\omega$ en $H(s)$, y antitransformamos usando las tablas de 
    transformadas de Fourier,
    \begin{align}

    	h(t)=&\boxed{0.22\Big[e^{-2.62t}-e^{-0.38t}\Big]u(t).}
    \end{align}

	\item 
El sistema es estable pues la ROC de $H(s)$ contiene al eje imaginario (existe $H(j\omega)$). Es también causal, 
pues la ROC de $H(s)$ es un semiplano derecho (la $h(t)$ resultante es causal).

\end{enumerate}

\item
La transformada de Laplace es una herramienta matemática extremadamente útil para analizar el funcionamiento de sistemas físicos como son los circuitos eléctricos. Se ve a continuación un ejemplo. 
\begin{enumerate}
\item
En este ejercicio se presenta un circuito $LR$, compuesto por un inductor y un resistor. Al describir el funcionamiento de este circuito en el dominio del tiempo, se obtiene una ecuación diferencial. Luego, aplicando las estrategias vistas anteriormente en este trabajo práctico, se usa la TL para resolver dicha ecuación diferencial, y de este modo obtener la respuesta correspondiente del circuito.

Se plantea en primer lugar una \textit{ecuación de malla} de la siguiente forma,
\begin{equation}
v_i(t)=v_L(t)+v_o(t),
\end{equation}
donde,\\
$v_i(t)$: voltaje de entrada (\textit{input}),\\
$v_L(t)$: voltaje en el inductor,\\
$v_o(t)$: voltaje de salida (\textit{output}), que en este caso es el voltaje en el resistor.

Los voltajes en el inductor y el resistor están dados por las leyes físicas correspondientes, a saber,
\begin{align}
v_L(t)&=L\frac{di(t)}{dt},\\
v_o(t)&=i(t)R.
\end{align}
La función de transferencia es la relación entre la salida y la entrada de un sistema en el dominio de la variable compleja $s$. Para llegar a dicha expresión, se puede operar de la siguiente manera,
\begin{align}
v_i(t)&=v_L(t)+v_o(t),\\
v_i(t)&=L\frac{di(t)}{dt}+v_o(t),\\
v_i(t)&=\frac{L}{R}\frac{dv_o(t)}{dt}+v_o(t).
\end{align}
Se aplica ahora transformada de Laplace,
\begin{align}
V_i(s)&=\frac{L}{R}sV_o(s)+Vo(s),\\
V_i(s)&=\left(\frac{L}{R}s+1\right)V_o(s),
\end{align}
y finalmente se despeja para encontrar la expresión de la función de transferencia,
\begin{align}
H(s)&=\frac{V_o(s)}{V_i(s)}=\frac{1}{\frac{L}{R}s+1},\\
&=\boxed{\frac{R/L}{s+R/L}}.
\end{align}
Para encontrar la respuesta al impulso, se hace $s=j\omega$ y se antitransforma la respuesta en frecuencia, 
$H(j\omega)$,
\begin{equation}
H(j\omega)=\frac{R/L}{j\omega+R/L},
\end{equation}
finalmente antitransformando,
\begin{equation}
h(t)=\boxed{\tfrac{R}{L}e^{-\frac{R}{L}t}u(t)}.
\end{equation}

\item
Se plantea de la misma forma que el ejercicio anterior resultando,
\begin{gather}
H(s)=\frac{V_o(s)}{V_i(s)}=\boxed{\frac{\frac{1}{RC}}{s+\frac{1}{RC}}}, \\
h(t)=\boxed{\tfrac{1}{RC}e^{-\frac{t}{RC}}u(t)}.
\end{gather}




\end{enumerate}

\end{enumerate}
